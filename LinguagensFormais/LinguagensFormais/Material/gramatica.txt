ProgArduino -> ListaComandos

ListaComandos ->
	DeclaraVar |
	Exp |
	If |
	For |
	While |
	DoWhile |
	ListaFunc |
	ListaComandos

DeclaraVar ->
	TipoVar id ListaVar |
	TipoVar id ListaVarTipo
ListaVar -> 
	, id ListaVar |
	{}
ListaVarTipo -> 
	, TipoVar id ListaVarTipo |
	{}
TipoVar ->
	int |
	long |
	float |
	byte

Exp -> 
	id () |
	Atribuicao |
	?

If ->
	if ( Exp ) { ListaComandos IfEnd
IfEnd ->
	} else { ListaComandos } |
	} else If |
	}

For ->
	for ( ListaDec; ListaAtrib; ListaAtrib ) { ListaComandos }
	
While ->
	while ( Exp ) { ListaComandos }

DoWhile ->
	do { ListaComandos } while ( Exp );

ListaFunc -> 
	TipoRetorno id ( ListaParm ) { ListaComandos Retorno }
ListaParm ->
	TipoVar id ListaParm |
	, TipoVar id ListaParm |
	{}
Retorno ->
	return Exp; |
	{}
TipoRetorno ->
	TipoVar |
	void
